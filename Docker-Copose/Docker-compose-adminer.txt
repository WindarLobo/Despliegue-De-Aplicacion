services:
  adminer:
    image: adminer
    restart: always
    ports:
      - 8080:8080
  db_postgres:
    image: postgres
    container_name: worldb-postgres
    volumes:
      - "pgdata:/var/lib/postgresql/data"
      - "./postgres:/docker-entrypoint-initdb.d"
    environment:
      - POSTGRES_USER=myuser
      - POSTGRES_PASSWORD=test1234
  nginx:
    image: nginx
    container_name: webserver
    restart: always
    ports:
      - "8888:80"
    volumes:
      - "./dist:/usr/share/nginx/html"
    networks:
      - testnet

  db:
    image: mysql
    container_name: worldb-mysql
    restart: always
    env_file: .env-secret
    ports:
      - "3306:3306"
    volumes:
      - "mysql-data:/var/lib/mysql"
      - "./world-db:/docker-entrypoint-initdb.d"
    networks:
      - testnet

networks:
  testnet:
    driver: bridge

volumes:
  mysql-data:
    driver: local
  pgdata:
    driver: local
